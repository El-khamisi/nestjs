
  deploy:
    runs-on: ubuntu-latest
    needs: [auto-format, init-env]

    steps:
      - name: steps to deploay
        uses: appleboy/ssh-action@v0.1.4
        with:
          username: root
          host: ${{ secrets.SSH_HOST }}
          key: ${{ secrets.SSH_KEY }}
          script:  |
            cd ~/nestjs
            git pull origin master
            npm i
            npm run build
            pm2 restart nestjs --update-env
      - name: Finished
        run: echo "### Hello "nestjs"! :rocket:" >> $GITHUB_STEP_SUMMARY
